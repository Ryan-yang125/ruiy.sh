3. ssh-agent

```bash
ssh-keygen -t ed25519 -C ruiyang0012@gmail.com
eval "$(ssh-agent -s)"
ssh-add ~/.ssh/id_ed25519
cat ~/.ssh/id_ed25519.pub
# ls ssh-agent process
ps aux | grep ssh-agent
ssh-add -l
ssh-add -D
```
